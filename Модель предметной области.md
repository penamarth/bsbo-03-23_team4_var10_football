entity Register
entity Match

    id

entity Ticket

entity TicketSession

entity Store

    date

Register "1" -down-> "0..*" Match : Создаётся
Register "1" -down- "1" Store : Обращение
Match "1" -down-> "*" Ticket : Создаётся
Match "1" -down-> "0..*" TicketSession : Создаётся
Ticket "1..*" -down- "1" Store : Записывается в продажу
TicketSession "0..*" -down- "1" Store : Записывается в продажу

Sale "1" -down- "1" Payment : Оплачивается
